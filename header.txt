    <script>
      document.addEventListener("DOMContentLoaded", function () {
        const parentItems = document.querySelectorAll(".parent-list li[data-category]");
        const subcategories = document.querySelectorAll(".subcategory");

        // Lắng nghe sự kiện hover vào mỗi mục cha
        parentItems.forEach((item) => {
          item.addEventListener("mouseenter", function () {
            // Ẩn tất cả các danh mục con
            subcategories.forEach((sub) => {
              sub.classList.remove("active");
            });

            // Lấy giá trị category của mục cha và hiển thị danh mục con tương ứng
            const cat = item.getAttribute("data-category");
            const sub = document.querySelector('.subcategory[data-category="' + cat + '"]');

            if (sub) {
              sub.classList.add("active"); // Hiển thị danh mục con tương ứng
            }
          });

          // Lắng nghe sự kiện rời chuột khỏi mục cha
          item.addEventListener("mouseleave", function () {
            // Không ẩn danh mục con ở đây nữa, vì khi chuột ra ngoài vẫn cần giữ danh mục con hiển thị
          });
        });

        // Lắng nghe sự kiện mouseenter cho các danh mục con để giữ chúng hiển thị khi di chuột vào danh mục con
        subcategories.forEach((sub) => {
          sub.addEventListener("mouseenter", function () {
            sub.classList.add("active"); // Giữ danh mục con hiển thị khi chuột vào
          });

          sub.addEventListener("mouseleave", function () {
            // Không ẩn ngay danh mục con khi chuột rời khỏi danh mục con
          });
        });

        // Ẩn tất cả danh mục con khi chuột rời khỏi vùng danh mục (category-hover-area)
        const hoverArea = document.querySelector(".category-hover-area");
        if (hoverArea) {
          hoverArea.addEventListener("mouseleave", function () {
            subcategories.forEach((sub) => {
              sub.classList.remove("active"); // Ẩn tất cả danh mục con khi chuột rời khỏi vùng hover
            });
          });
        }
      });
    </script>
